// Benchmarks for POR
// 5 IN.OUT in parallel

// Channels
free c0.
free c1.
free c2.
free c3.
free c4.

// Public constant
free ok.
let P =
new n0;
new n1;
new n2;
new n3;
new n4;
(( in(c0, x); if x = ok then out(c0, n0) ) |
( in(c1, x); if x = ok then out(c1, n1) ) |
( in(c2, x); if x = ok then out(c2, n2) ) |
( in(c3, x); if x = ok then out(c3, n3) ) |
( in(c4, x); if x = ok then out(c4, n4) )
).

query trace_equiv(P,P).
